116 void odometryUpdate()
117 {
118 auto position = vehicle_local_position_->positionNed();
119 auto velocity = vehicle_local_position_->velocityNed();
120 auto attitude = vehicle_attitude_->attitude();
121 auto angular_velocity = vehicle_angular_velocity_->angularVelocityFrd();
122 if (!mode_) {
123 RCLCPP_ERROR(node_.get_logger(), "Mode not initialized");
124 return;
125 }
126 this->mode_->setCurrentOdometry(
127 eigenNedToTf2Nwu(position),
128 eigenNedToTf2Nwu(attitude),
129 eigenNedToTf2Nwu(velocity),
130 eigenNedToTf2Nwu(angular_velocity));
131 }
