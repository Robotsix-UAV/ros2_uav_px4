57 template<typename ModeT>
58 void AttitudeThrustMode<ModeT>::updateSetpoint([[maybe_unused]] float dt)
59 {
60  odometryUpdate();
61  double elapsed_time = (node_.now() - time_init_).seconds();
62  AttitudeThrust control_inputs = mode_.triggerMode(elapsed_time);
63  // Set the attitude setpoint
64  // Conversion of the thrust
65  double thrust_constant_coefficient, thrust_linear_coefficient, thrust_quadratic_coefficient;
66  getParameter("px4.thrust_constant_coefficient", thrust_constant_coefficient);
67  getParameter("px4.thrust_linear_coefficient", thrust_linear_coefficient);
68  getParameter("px4.thrust_quadratic_coefficient", thrust_quadratic_coefficient);
69  float thrust = control_inputs.thrust;
70  float normalized_thrust = 0.0;
71  // Find the normalized with thrust = c + l * t_n + q * t_n^2
72  float discriminant = thrust_linear_coefficient * thrust_linear_coefficient -
73  4 * thrust_quadratic_coefficient * (thrust_constant_coefficient - thrust);
74  if (discriminant < 0) {
75  RCLCPP_WARN(
76  node_.get_logger(), "[Attitude Thrust Mode] Negative discriminant in thrust conversion");
77  } else {
78  normalized_thrust = (-thrust_linear_coefficient + std::sqrt(discriminant)) /
79  (2 * thrust_quadratic_coefficient);
80  }
81  const Eigen::Vector3f thrust_sp{0.0f, 0.0f, static_cast<float>(-normalized_thrust)};
82  const Eigen::Quaternionf attitude_sp = tf2FwuToEigenNed(control_inputs.orientation);
83  attitude_setpoint_->update(attitude_sp, thrust_sp);
84 }
