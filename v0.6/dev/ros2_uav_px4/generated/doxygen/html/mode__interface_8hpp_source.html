116 void odometryUpdate()
117 {
118 uav_cpp::types::Odometry odometry;
119 auto position = vehicle_local_position_->positionNed();
120 auto velocity = vehicle_local_position_->velocityNed();
121 auto attitude = vehicle_attitude_->attitude();
122 auto angular_velocity = vehicle_angular_velocity_->angularVelocityFrd();
123 odometry.position = eigenNedToTf2Nwu(position);
124 odometry.velocity = eigenNedToTf2Nwu(velocity);
125 odometry.attitude = eigenNedToTf2Nwu(attitude);
126 odometry.angular_velocity = eigenNedToTf2Nwu(angular_velocity);
127 if (!pipeline_) {
128 RCLCPP_ERROR(node_.get_logger(), "ControlPipeline not initialized");
129 return;
130 }
131 this->pipeline_->setCurrentOdometry(odometry);
132 }
