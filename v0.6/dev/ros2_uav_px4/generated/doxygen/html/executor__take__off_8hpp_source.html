77 void runState(State state)
78 {
79 switch (state) {
80 case State::ARM:
81 RCLCPP_INFO(node().get_logger(), "[TakeOff executor] Arming");
82 arm(
83 [this](px4_ros2::Result result)
84 {
85 if (result == px4_ros2::Result::Success) {
86 runState(State::TAKEOFF);
87 }
88 });
89 break;
90 case State::TAKEOFF:
91 RCLCPP_INFO(node().get_logger(), "[TakeOff executor] Taking off");
92 double altitude;
93 getParameter("takeoff.altitude", altitude);
94 takeoff(
95 [this](px4_ros2::Result result)
96 {
97 if (result == px4_ros2::Result::Success) {
98 runState(State::OWNED_MODE);
99 }
100 },
101 altitude, 0);
102 break;
103
104 case State::OWNED_MODE:
105 RCLCPP_INFO(node().get_logger(), "[TakeOff executor] Owned mode");
106 scheduleMode(
107 ownedMode().id(), [](px4_ros2::Result) {return;});
108 break;
109 }
110 }
111};
