43template<typename PipelineT>
44concept DerivedFromUavCppPipeline = requires(PipelineT pipeline)

46 [] < typename ... T > (uav_cpp::pipelines::ControlPipeline<T...> &) {} (pipeline);

57public:
64 ModeInterface(const ModeBase::Settings & mode_settings, rclcpp::Node & node)
65 : ModeBase(node, mode_settings),
66 ParamContainer(),
67 node_(node)
68 {
69 this->createMode();
70 this->addChildContainer(this->pipeline_.get());
71 vehicle_local_position_ = std::make_shared<px4_ros2::OdometryLocalPosition>(*this);
72 vehicle_angular_velocity_ = std::make_shared<px4_ros2::OdometryAngularVelocity>(*this);
73 vehicle_attitude_ = std::make_shared<px4_ros2::OdometryAttitude>(*this);
74 addRequiredParameter<bool>("disturbance_observer.enable");
75 // Add a subscription to the disturbance topic
76 disturbance_sub_ = node_.create_subscription<ros2_uav_interfaces::msg::Disturbance>(
77 "disturbance", 1, [this](const ros2_uav_interfaces::msg::Disturbance::SharedPtr msg) {
78 bool disturbance_flag;
79 getParameter("disturbance_observer.enable", disturbance_flag);
80 if (!disturbance_flag) {
81 return;
82 }
83 uav_cpp::types::DisturbanceCoefficientsStamped disturbance_coefficients;
84 disturbance_coefficients = uav_ros2::utils::convert(*msg);
85 pipeline_->setDisturbanceCoefficients(disturbance_coefficients);
86 });
87 auto mode_name = mode_settings.name;
88 std::replace(mode_name.begin(), mode_name.end(), ' ', '_');
89 }

130 void odometryUpdate()
131 {
132 uav_cpp::types::OdometryStamped odometry;
133 auto position = vehicle_local_position_->positionNed();
134 auto velocity = vehicle_local_position_->velocityNed();
135 auto attitude = vehicle_attitude_->attitude();
136 auto angular_velocity = vehicle_angular_velocity_->angularVelocityFrd();
137 odometry.frame_id = uav_cpp::enums::Frame::ODOM;
138 odometry.timestamp = std::chrono::microseconds(vehicle_local_position_->last().timestamp);
139 odometry.position = NedToNwu(position.cast<double>());
140 odometry.velocity = NedToNwu(velocity.cast<double>());
141 odometry.attitude = NedToNwu(attitude.cast<double>());
142 odometry.angular_velocity = NedToNwu(angular_velocity.cast<double>());
143 if (!pipeline_) {
144 UAVCPP_ERROR("ControlPipeline not initialized");
145 return;
146 }
147 this->pipeline_->setCurrentOdometry(odometry);
148 }
