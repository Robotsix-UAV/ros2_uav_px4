39uav_cpp::types::Frame convert(const std::string & frame_id)
40{
41 if (frame_id.find("odom") != std::string::npos) {
42 return uav_cpp::types::Frame::ODOM;
43 } else if (frame_id.find("base_link") != std::string::npos) {
44 return uav_cpp::types::Frame::BASE_LINK;
45 } else {
46 return uav_cpp::types::Frame::UNKNOWN;
47 }

72uav_cpp::types::PoseHeadingStamped convert(
73 const ros2_uav_interfaces::msg::PoseHeading & msg)
74{
75 uav_cpp::types::PoseHeadingStamped setpoint;
76 setpoint.timestamp =
77 std::chrono::nanoseconds{static_cast<uint64_t>(msg.header.stamp.sec) * 1000000000 +
78 static_cast<uint64_t>(msg.header.stamp.nanosec)};
79 setpoint.frame_id = convert(msg.header.frame_id);
80 setpoint.position = Eigen::Vector3d(msg.position.x, msg.position.y, msg.position.z);
81 setpoint.velocity = Eigen::Vector3d(msg.velocity.x, msg.velocity.y, msg.velocity.z);
82 setpoint.heading = msg.heading;
83 return setpoint;
84}

110ros2_uav_interfaces::msg::Disturbance convert(
111 const uav_cpp::types::DisturbanceCoefficientsStamped & disturbance)
112{
113 ros2_uav_interfaces::msg::Disturbance msg;
114 msg.header.stamp.sec = disturbance.timestamp.count() / 1000000000;
115 msg.header.stamp.nanosec = disturbance.timestamp.count() % 1000000000;
116 msg.header.frame_id = convert(disturbance.frame_id);
117 msg.constant.x = disturbance.constant.vector.x();
118 msg.constant.y = disturbance.constant.vector.y();
119 msg.constant.z = disturbance.constant.vector.z();
120 msg.proportional.x = disturbance.proportional.vector.x();
121 msg.proportional.y = disturbance.proportional.vector.y();
122 msg.proportional.z = disturbance.proportional.vector.z();
123 return msg;
124}

131uav_cpp::types::AttitudeThrustStamped convert(

133{

135 att_thrust.timestamp = std::chrono::microseconds{msg.timestamp};
136 att_thrust.attitude = Eigen::Quaterniond(msg.q_d[0], msg.q_d[1], -msg.q_d[2], -msg.q_d[3]);

146uav_cpp::types::OdometryStamped convert(const px4_msgs::msg::VehicleOdometry & msg)
147{
148 uav_cpp::types::OdometryStamped odometry;

150 odometry.position = Eigen::Vector3d(msg.position[0], -msg.position[1], -msg.position[2]);
151 odometry.velocity = Eigen::Vector3d(msg.velocity[0], -msg.velocity[1], -msg.velocity[2]);

173uav_cpp::types::PoseSpeedVectorStamped convert(
174 const ros2_uav_interfaces::msg::WaypointList & msg)
175{
176 uav_cpp::types::PoseSpeedVectorStamped pose_speed_vector;
177 pose_speed_vector.timestamp =
178 std::chrono::nanoseconds{static_cast<uint64_t>(msg.header.stamp.sec) * 1000000000 +
179 static_cast<uint64_t>(msg.header.stamp.nanosec)};
180 pose_speed_vector.frame_id = convert(msg.header.frame_id);
181 for (const auto & waypoint : msg.waypoints) {
182 uav_cpp::types::PoseSpeed pose_speed;
183 pose_speed.position = Eigen::Vector3d(
184 waypoint.position.x, waypoint.position.y,
185 waypoint.position.z);
186 pose_speed.speed = waypoint.speed;
187 pose_speed_vector.waypoints.push_back(pose_speed);
188 }
189 return pose_speed_vector;
190}
