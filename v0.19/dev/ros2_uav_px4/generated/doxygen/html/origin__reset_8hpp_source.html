30class OriginReset

32public:
38 OriginReset(
39 rclcpp::Node & node,
40 int target_system_id)
41 : target_system_id_(target_system_id),
42 node_(node)
43 {
44 vehicle_command_pub = node.create_publisher<px4_msgs::msg::VehicleCommand>(
45 "fmu/in/vehicle_command", 1);
46 vehicle_global_position_sub = node.create_subscription<px4_msgs::msg::VehicleGlobalPosition>(
47 "fmu/out/vehicle_global_position", rclcpp::QoS(1).best_effort(),
48 [this](const px4_msgs::msg::VehicleGlobalPosition::SharedPtr msg) {
49 current_lat = msg->lat;
50 current_lon = msg->lon;
51 current_alt = msg->alt;
52 global_position_received = true;
53 });
54 }
55
60 bool resetOrigin()
61 {
62 if (global_position_received) {
63 px4_msgs::msg::VehicleCommand msg;
64 msg.timestamp = round(node_.now().nanoseconds() / 1000.0);
65 msg.target_system = target_system_id_;
66 msg.target_component = 1;
67 msg.source_system = 255;
68 msg.from_external = true;
69 msg.param5 = current_lat;
70 msg.param6 = current_lon;
71 msg.param7 = current_alt;
72 msg.command = px4_msgs::msg::VehicleCommand::VEHICLE_CMD_SET_GPS_GLOBAL_ORIGIN;
73 vehicle_command_pub->publish(msg);
74 UAVCPP_INFO("Global position origin reset");
75 return true;
76 }
77 return false;
78 }
79
80private:
81 int target_system_id_;
82 rclcpp::Publisher<px4_msgs::msg::VehicleCommand>::SharedPtr vehicle_command_pub;
84 rclcpp::Subscription<px4_msgs::msg::VehicleGlobalPosition>::SharedPtr vehicle_global_position_sub;
86 rclcpp::Node & node_;
87 bool global_position_received = false;
88 float current_lat = 0;
89 float current_lon = 0;
90 float current_alt = 0;
91};
