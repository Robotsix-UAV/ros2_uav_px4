128 void odometryUpdate()
129 {
130 uav_cpp::types::Odometry odometry;
131 auto position = vehicle_local_position_->positionNed();
132 auto velocity = vehicle_local_position_->velocityNed();
133 auto attitude = vehicle_attitude_->attitude();
134 auto angular_velocity = vehicle_angular_velocity_->angularVelocityFrd();
135 odometry.position = eigenNedToTf2Nwu(position);
136 odometry.velocity = eigenNedToTf2Nwu(velocity);
137 odometry.attitude = eigenNedToTf2Nwu(attitude);
138 odometry.angular_velocity = eigenNedToTf2Nwu(angular_velocity);
139 if (!pipeline_) {
140 RCLCPP_ERROR(node_.get_logger(), "ControlPipeline not initialized");
141 return;
142 }
143 this->pipeline_->setCurrentOdometry(odometry);
144 }
