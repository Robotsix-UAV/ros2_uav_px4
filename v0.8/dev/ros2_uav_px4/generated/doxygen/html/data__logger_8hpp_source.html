33class DataLogger
34{
35public:
36 DataLogger()
37 {
38 std::string ros_log_directory = getRosLogDirectory();
39 // Create a directory with the current date and time
40 std::time_t t = std::time(nullptr);
41 std::tm tm = *std::localtime(&t);
42 std::stringstream ss;
43 ss << std::put_time(&tm, "%Y-%m-%d-%H-%M-%S");
44 log_folder_ = ros_log_directory + ss.str();
45 if (mkdir(log_folder_.c_str(), 0777) == -1) {
46 UAVCPP_ERROR("[DataLogger] Error creating log directory: {}", log_folder_);
47 }
48 }
49
50 void logToFile(
51 std::string file_name, std::vector<std::chrono::nanoseconds> timestamps,
52 std::vector<std::string> labels, std::vector<std::vector<double>> data)
53 {
54 // Check if the number of timestamps is the same as the number of data points
55 if (timestamps.size() != data.size()) {
56 UAVCPP_ERROR(
57 "[DataLogger] Number of timestamps ({}) is different from the number of data points ({})",
58 timestamps.size(), data.size());
59 return;
60 }
61 std::string log_file_path = log_folder_ + "/" + file_name;
62 std::ofstream log_file(log_file_path, std::ios::out | std::ios::app);
63 size_t timestamp_index = 0;
64 if (log_file.is_open()) {
65 log_file << "timestamp";
66 // Write the labels
67 for (size_t i = 0; i < labels.size(); ++i) {
68 log_file << ",";
69 log_file << labels[i];
70 }
71 log_file << std::endl;
72 for (const auto & row : data) {
73 if (timestamp_index < timestamps.size()) {
74 log_file << timestamps[timestamp_index].count();
75 timestamp_index++;
76 }
77 for (size_t i = 0; i < row.size(); ++i) {
78 log_file << ",";
79 log_file << row[i];
80 }
81 log_file << std::endl;
82 }
83 log_file.close();
84 } else {
85 UAVCPP_ERROR("[DataLogger] Error opening log file: {}", log_file_path);
86 }
87 }
88
89private:
90 std::string log_folder_;
91
92 std::string getRosLogDirectory()
93 {
94 auto ros_home = std::getenv("ROS_HOME");
95 std::string default_log_dir;
96 if (ros_home) {
97 default_log_dir = std::string(ros_home) + "/log/";
98 } else {
99 std::string home = std::getenv("HOME") ? std::getenv("HOME") : "";
100 default_log_dir = home + "/.ros/log/";
101 }
102 UAVCPP_INFO("[DataLogger] Using ROS log directory: {}", default_log_dir);
103 return default_log_dir;
104 }
105};
