31class ButterworthFilter
32{
33public:
40 ButterworthFilter(double cutoff_frequency, double sampling_frequency)
41 : cutoff_(cutoff_frequency), fs_(sampling_frequency)
42 {
43 computeCoefficients();
44 }
45
52 std::vector<double> filtfilt(const std::vector<double> & input)
53 {
54 std::vector<double> filtered = filter(input);
55 std::reverse(filtered.begin(), filtered.end());
56 filtered = filter(filtered);
57 std::reverse(filtered.begin(), filtered.end());
58 return filtered;
59 }
60
67 std::vector<double> filter(const std::vector<double> & input)
68 {
69 std::vector<double> output(input.size(), 0.0);
70 for (size_t i = 0; i < input.size(); i++) {
71 output[i] = a0 * input[i];
72 if (i > 0) {
73 output[i] += a1 * input[i - 1] - b1 * output[i - 1];
74 }
75 if (i > 1) {
76 output[i] += a2 * input[i - 2] - b2 * output[i - 2];
77 }
78 }
79 return output;
80 }
81
82private:
83 double cutoff_; //< Cutoff frequency
84 double fs_; //< Sampling frequency
85 double a0, a1, a2, b1, b2; //< Filter coefficients
86
90 void computeCoefficients()
91 {
92 double wc = tan(M_PI * cutoff_ / fs_);
93 double k1 = sqrt(2.0) * wc;
94 double k2 = wc * wc;
95 double norm = 1 / (1 + k1 + k2);
96
97 a0 = k2 * norm;
98 a1 = 2 * a0;
99 a2 = a0;
100 b1 = 2 * (k2 - 1) * norm;
101 b2 = (1 - k1 + k2) * norm;
102 }
103};

108class DerivativeFilter
109{
110public:
117 DerivativeFilter(double cutoff_frequency, double sampling_frequency)
118 : filter_(cutoff_frequency, sampling_frequency), fs_(sampling_frequency) {}
119
125 void addData(double value)
126 {
127 data_.push_back(value);
128 }
129
135 std::vector<double> computeDerivative()
136 {
137 if (data_.size() < 3) {return {};}
138 std::vector<double> derivatives(data_.size(), 0.0);
139 double dt = 1.0 / fs_;
140 for (size_t i = 1; i < data_.size() - 1; ++i) {
141 derivatives[i] = (data_[i + 1] - data_[i - 1]) / (2 * dt);
142 }
143 return derivatives;
144 }
145
151 std::vector<double> process()
152 {
153 auto derivatives = computeDerivative();
154 if (derivatives.empty()) {return {};}
155 return filter_.filtfilt(derivatives);
156 }
157
164 void process(std::vector<double> & derivative, std::vector<double> & derivative_filtered)
165 {
166 derivative = computeDerivative();
167 if (derivative.empty()) {return;}
168 derivative_filtered = filter_.filtfilt(derivative);
169 }
170
171private:
172 ButterworthFilter filter_; //< Butterworth filter
173 double fs_; //< Sampling frequency
174 std::vector<double> data_; //< Input data
175};

180class DerivativeFilter3D
181{
182public:
189 DerivativeFilter3D(double cutoff_frequency, double sampling_frequency)
190 : filter_x_(cutoff_frequency, sampling_frequency),
191 filter_y_(cutoff_frequency, sampling_frequency),
192 filter_z_(cutoff_frequency, sampling_frequency),
193 fs_(sampling_frequency) {}
194
199 void addData(const tf2::Vector3 & value)
200 {
201 filter_x_.addData(value.x());
202 filter_y_.addData(value.y());
203 filter_z_.addData(value.z());
204 }
205
211 std::vector<tf2::Vector3> process()
212 {
213 auto x = filter_x_.process();
214 auto y = filter_y_.process();
215 auto z = filter_z_.process();
216 if (x.empty() || y.empty() || z.empty()) {return {};}
217 std::vector<tf2::Vector3> derivatives;
218 for (size_t i = 0; i < x.size(); ++i) {
219 derivatives.emplace_back(x[i], y[i], z[i]);
220 }
221 return derivatives;
222 }
223
230 void process(
231 std::vector<tf2::Vector3> & derivative,
232 std::vector<tf2::Vector3> & derivative_filtered)
233 {
234 std::vector<double> x, y, z, xf, yf, zf;
235 filter_x_.process(x, xf);
236 filter_y_.process(y, yf);
237 filter_z_.process(z, zf);
238 if (x.empty() || y.empty() || z.empty() || xf.empty() || yf.empty() || zf.empty()) {return;}
239 derivative.clear();
240 for (size_t i = 0; i < x.size(); ++i) {
241 derivative.emplace_back(x[i], y[i], z[i]);
242 }
243 derivative_filtered.clear();
244 for (size_t i = 0; i < xf.size(); ++i) {
245 derivative_filtered.emplace_back(xf[i], yf[i], zf[i]);
246 }
247 }
248
249private:
250 DerivativeFilter filter_x_; //< Derivative filter for the x-axis
251 DerivativeFilter filter_y_; //< Derivative filter for the y-axis
252 DerivativeFilter filter_z_; //< Derivative filter for the z-axis
253 double fs_; //< Sampling frequency
254};
