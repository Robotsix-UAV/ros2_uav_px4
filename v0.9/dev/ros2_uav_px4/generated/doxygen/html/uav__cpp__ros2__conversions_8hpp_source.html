35uav_cpp::pipelines::PoseHeading convertFromRosMsg(const ros2_uav_interfaces::msg::PoseHeading & msg)
36{
37 uav_cpp::pipelines::PoseHeading setpoint;
38 setpoint.timestamp =
39 std::chrono::nanoseconds{static_cast<uint64_t>(msg.header.stamp.sec) * 1000000000 +
40 static_cast<uint64_t>(msg.header.stamp.nanosec)};
41 setpoint.frame_id = msg.header.frame_id;
42 setpoint.position = tf2::Vector3(msg.position.x, msg.position.y, msg.position.z);
43 setpoint.velocity = tf2::Vector3(msg.velocity.x, msg.velocity.y, msg.velocity.z);
44 setpoint.heading = msg.heading;
45 return setpoint;
46}

53uav_cpp::pipelines::AttitudeThrust convertFromPx4Msg(
54 const px4_msgs::msg::VehicleAttitudeSetpoint & msg)
55{
56 uav_cpp::pipelines::AttitudeThrust att_thrust;
57 att_thrust.timestamp =
58 std::chrono::microseconds{msg.timestamp};
59 att_thrust.orientation = tf2::Quaternion(msg.q_d[1], -msg.q_d[2], -msg.q_d[3], msg.q_d[0]);
60 att_thrust.thrust = -msg.thrust_body[2];
61 return att_thrust;
62}

69uav_cpp::pipelines::Odometry convertFromPx4Msg(const px4_msgs::msg::VehicleOdometry & msg)
70{
71 uav_cpp::pipelines::Odometry odometry;
72 odometry.timestamp = std::chrono::microseconds{msg.timestamp};
73 odometry.odometry.position = tf2::Vector3(msg.position[0], -msg.position[1], -msg.position[2]);
74 odometry.odometry.velocity = tf2::Vector3(msg.velocity[0], -msg.velocity[1], -msg.velocity[2]);
75 odometry.odometry.orientation = tf2::Quaternion(msg.q[1], -msg.q[2], -msg.q[3], msg.q[0]);
76 odometry.odometry.angular_velocity = tf2::Vector3(
77 msg.angular_velocity[0],
78 -msg.angular_velocity[1],
79 -msg.angular_velocity[2]);
80 return odometry;
81}
