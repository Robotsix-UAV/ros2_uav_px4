79  void publishCoordinates(const std::vector<Coordinate::SharedPtr> & coordinates)
80  {
81  float final_time;
82  std::vector<double> sample_times;
83  for (const auto & coordinate : coordinates) {
84  ros2_uav_interfaces::msg::Coordinate coordinate_msg;
85  ros2_uav_interfaces::msg::FloatArray float_array_msg;
86  coordinate_msg.name = coordinate->getName();
87  final_time = coordinate->getFinalTime();
88  sample_times.clear();
89  for (double sample_time = 0.0; sample_time <= final_time; sample_time += 0.01) {
90  sample_times.push_back(sample_time);
91  }
92  for (uint8_t derivative = 0; derivative <= coordinate->getMaxDerivativeOrder();
93  ++derivative)
94  {
95  if (derivative < coordinate->getMinDerivativeOrder()) {
96  coordinate_msg.derivatives.push_back(ros2_uav_interfaces::msg::FloatArray());
97  } else {
98  coordinate->getTrajectory(sample_times, derivative, float_array_msg.data);
99  coordinate_msg.derivatives.push_back(float_array_msg);
100  }
101  }
102  coordinate_msg.frame_id = mode_.getControllerReferenceFrame();
103  coordinate_msg.timestamps = sample_times;
104  coordinate_publisher_->publish(coordinate_msg);
105  }
106  }
