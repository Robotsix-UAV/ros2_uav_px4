37template<DerivedFromRatesThrustMode ModeT>
38void RatesThrustMode<ModeT>::updateSetpoint([[maybe_unused]] float dt)
39{
40 this->odometryUpdate();
41 auto elapsed_time = (this->node_.now()).nanoseconds();
42 auto control_inputs = this->pipeline_->execute(std::chrono::nanoseconds(elapsed_time));
43 const Eigen::Vector3f thrust_sp{0.0f, 0.0f, -static_cast<float>(control_inputs.thrust)};
44 const Eigen::Vector3f rates_sp = tf2FwuToEigenNed(control_inputs.rates);
45 rates_setpoint_->update(rates_sp, thrust_sp);
46}
