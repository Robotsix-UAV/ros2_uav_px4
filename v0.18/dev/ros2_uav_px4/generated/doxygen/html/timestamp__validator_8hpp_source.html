34class TimestampValidator : public uav_cpp::logger::LogTagHolder
35{
36public:
43 explicit TimestampValidator(int64_t tolerance_ns = 1000000000, int max_outliers = 10)
44 : uav_cpp::logger::LogTagHolder("Timestamp Validator"),
45 previous_timestamp_ns_(0),
46 tolerance_ns_(tolerance_ns),
47 outlier_count_(0),
48 max_outliers_(max_outliers),
49 is_first_timestamp_(true)
50 {
51 previous_time_point_ = std::chrono::steady_clock::now();
52 }
53
60 bool isValidTimestamp(int64_t timestamp_ns)
61 {
62 auto now = std::chrono::steady_clock::now();
63
64 if (is_first_timestamp_) {
65 // Always accept the first timestamp
66 previous_timestamp_ns_ = timestamp_ns;
67 previous_time_point_ = now;
68 is_first_timestamp_ = false;
69 outlier_count_ = 0;
70 return true;
71 }
72
73 // Check if the timestamp is monotonic
74 if (timestamp_ns < previous_timestamp_ns_) {
75 // Timestamp decreased, invalid
76 outlier_count_++;
77 if (outlier_count_ >= max_outliers_) {
78 resetReference();
79 }
80 UAVCPP_WARN_TAG(this, "Timestamp decreased, invalid");
81 return false;
82 }
83
84 // Compute expected timestamp based on previous timestamp and elapsed time
85 int64_t elapsed_ns = std::chrono::duration_cast<std::chrono::nanoseconds>(
86 now - previous_time_point_).count();
87 int64_t expected_timestamp_ns = previous_timestamp_ns_ + elapsed_ns;
88
89 int64_t diff_ns = timestamp_ns - expected_timestamp_ns;
90
91 if (std::abs(diff_ns) > tolerance_ns_) {
92 // Difference is larger than tolerance, invalid
93 outlier_count_++;
94 if (outlier_count_ >= max_outliers_) {
95 resetReference();
96 }
97 UAVCPP_WARN_TAG(this, "Timestamp too far from expected, invalid");
98 UAVCPP_WARN_TAG(
99 this, "Expected: {}, Received: {}", expected_timestamp_ns,
100 timestamp_ns);
101 return false;
102 }
103
104 // Timestamp is valid
105 previous_timestamp_ns_ = timestamp_ns;
106 previous_time_point_ = now;
107 outlier_count_ = 0; // Reset outlier count
108 return true;
109 }
110
114 void resetReference()
115 {
116 previous_timestamp_ns_ = 0;
117 outlier_count_ = 0;
118 previous_time_point_ = std::chrono::steady_clock::now();
119 is_first_timestamp_ = true;
120 }
121
122private:
123 int64_t previous_timestamp_ns_;
124 int64_t tolerance_ns_;
125 int outlier_count_;
126 int max_outliers_;
127 bool is_first_timestamp_;
128 std::chrono::steady_clock::time_point previous_time_point_;
130};
